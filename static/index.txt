    <div class="container">
        <h1>Поиск с помощью ИИ</h1>
        
        <form id="search-form">
            <input type="text" id="search-input" placeholder="Введите ваш запрос..." required>
            <button type="submit" id="search-button">Найти</button>
        </form>

        <div id="results">
            <!-- Результаты поиска будут появляться здесь -->
        </div>
    </div>

    <script>
        // Находим наши элементы на странице
        const searchForm = document.getElementById('search-form');
        const searchInput = document.getElementById('search-input');
        const resultsDiv = document.getElementById('results');

        // Добавляем обработчик на отправку формы
        searchForm.addEventListener('submit', async (event) => {
            // Предотвращаем стандартное поведение формы (перезагрузку страницы)
            event.preventDefault(); 
            
            const queryText = searchInput.value;
            
            // Показываем пользователю, что что-то происходит
            resultsDiv.innerHTML = '<p class="loading">Идет поиск...</p>';

            try {
                // !!! ВАЖНО: Замените этот URL на адрес вашего FastAPI сервера !!!
                const apiUrl = 'http://127.0.0.1:8000/api/ask';
                
                // Отправляем POST-запрос с помощью fetch
                const response = await fetch(apiUrl, {
                    method: 'POST',
                    headers: {
                        'Content-Type': 'application/json', // Говорим серверу, что отправляем JSON
                    },
                    // Преобразуем наш объект в JSON-строку
                    body: JSON.stringify({ query: queryText }), 
                });

                // Проверяем, что ответ успешный (статус 200-299)
                if (!response.ok) {
                    throw new Error(`Ошибка сервера: ${response.status}`);
                }

                // Преобразуем ответ от сервера из JSON в объект
                const data = await response.json();

                // Отображаем полученные результаты
                displayResults(data.results);

            } catch (error) {
                console.error('Ошибка при выполнении запроса:', error);
                resultsDiv.innerHTML = `<p style="color: red;">Не удалось выполнить поиск. Убедитесь, что сервер запущен и доступен. Ошибка: ${error.message}</p>`;
            }
        });

        // Функция для отображения результатов на странице
        function displayResults(results) {
            resultsDiv.innerHTML = ''; // Очищаем старые результаты

            // Если есть результаты, перебираем их и создаем HTML для каждого
            if (results && results.length > 0) {
                results.forEach(item => {
                    const resultItem = document.createElement('div');
                    resultItem.className = 'result-item';
                    // Адаптируйте эту разметку под структуру вашего ответа от API
                    resultItem.innerHTML = `
                        <h3>${item.title || 'Без заголовка'}</h3>
                        <p>Релевантность: ${(item.score * 100).toFixed(2)}%</p>
                    `;
                    resultsDiv.appendChild(resultItem);
                });
            } else {
                resultsDiv.innerHTML = '<p>Ничего не найдено.</p>';
            }
        }
    </script>
